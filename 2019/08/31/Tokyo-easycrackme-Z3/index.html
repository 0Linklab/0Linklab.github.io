def sum_exp(li):
    result = 0

def count_cipin(li):
    result = [0 for _ in range(0,16)]
    F = '0123456789abcdef'
    for l in F:
        for m in li:
            if m == ord(l):
                result[int(l)] +=1
    return result


def count_add21(li):
    result = []
    F = 0
    for m in range(0,8):
        for n in range(0,4):
            F += li[4*m+n]
        result.append(F)
    return result
