string sort_string(string it){
    char arrchr[it.size()];
    string tmp ="";
    for(int i=0;i<it.size();++i){
        arrchr[i] = it.at(i);
    }
    sort(&arrchr[0],&arrchr[it.size()]);
    for(int i=0;i<it.size();++i){
        tmp += arrchr[i];
    }
    return tmp;
}

    set<string> tmp;

    return tmp;
}
