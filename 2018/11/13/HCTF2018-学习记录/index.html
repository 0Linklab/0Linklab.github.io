#challenge.py
from Crypto.Util.strxor import strxor
import base64
import random


def enc(data, key):
    key = (key * (len(data) / len(key) + 1))[:len(data)]
    return strxor(data, key)


poem = open('poem.txt', 'r').read()
flag = "hctf{xxxxxxxxxxx}"

with open('cipher.txt', 'w') as f:
    f.write(base64.b64encode(enc(poem, flag[5:-1])))
    f.close()

        通过简单分析其逻辑，可以知道xor_key就是flag花括号中的内容，通过该xor_key对poem.txt中的内容进行加密，并且需要对应长度。因为poem.txt中是一首诗，诗中可能会出现字母，数字，引号，换行等等合法字符，由于我们只有密文，并且知道xor的规律，是xor_key的长度来决定的，那么就可以通过遍历加上简单的判断，写出如下代码(这是最简单的解法，参照电子科大CNSS战队的wp ）：

#exp.py
import base64

#检测是否为诗歌正常字符
def invalid(x):
    if chr(x) in '{}[]@#%^*=+':
        return True
    #换行符为正常字符
    if x == 10:
        return False
    if x <= 31:
        return True
    return False

cipher = base64.b64decode(open('cipher.txt', 'r').read())
# 可能的xor_key长度
for L in range(1, 32):
    c = []
    cc = []
    #对xor_key的每一位进行枚举
    for i in range(L):
        t = []
        for cand in range(32, 128):
            flag = True
            #步长为xor_key长度，进行测试
            for j in range(i, len(cipher), L):
                tmp = cand ^ cipher[j]
                #若不为正常字符，则将flag标志位置为False,不会将该ascii作为可能的xor_key字符
                if invalid(tmp):
                    flag = False
                    break
            #将符合条件的ascii对应的字符添加到对应xor_key字符位置的列表
            if flag:
                t.append(chr(cand))
        c.append(len(t))
        cc.append(t)
    if 0 not in c:
        print(L, c)
        for i in range(L):
            print('\t', i, cc[i])

#.sage文件
def short_pad_attack(c1, c2, e, n):
    PRxy.<x,y> = PolynomialRing(Zmod(n))
    PRx.<xn> = PolynomialRing(Zmod(n))
    PRZZ.<xz,yz> = PolynomialRing(Zmod(n))

    g1 = x^e - c1
    g2 = (x+y)^e - c2

    q1 = g1.change_ring(PRZZ)
    q2 = g2.change_ring(PRZZ)

    h = q2.resultant(q1)
    h = h.univariate_polynomial()
    h = h.change_ring(PRx).subs(y=xn)
    h = h.monic()

    kbits = n.nbits()//(2*e*e)
    diff = h.small_roots(X=2^kbits, beta=0.5)[0]  # find root < 2^kbits with factor >= n^0.5

    return diff

def related_message_attack(c1, c2, diff, e, n):
    PRx.<x> = PolynomialRing(Zmod(n))
    g1 = x^e - c1
    g2 = (x+diff)^e - c2

    def gcd(g1, g2):
        while g2:
            g1, g2 = g2, g1 % g2
        return g1.monic()

    return -gcd(g1, g2)[0]

def main():
    # check argv
    if len(sys.argv) != 2:
        exit(1)

    # get argv
    pcapFilePath = sys.argv[1]

    # get data of pcap
    os.system("tshark -r %s -T fields -e usb.capdata > %s" % (pcapFilePath, DataFileName))

    # read data
    with open(DataFileName, "r") as f:
        for line in f:
            presses.append(line[0:-1])
    # handle
    result = ""
    for press in presses:
        Bytes = press.split(":")
        if Bytes[0] == "00":
            if Bytes[2] != "00":
                result += normalKeys[Bytes[2]]
        elif Bytes[0] == "02": # shift key is pressed.

                result += shiftKeys[Bytes[2]]
        elif Bytes[0] == "01":
            if Bytes[2] != "00":
                result += ("Ctrl+" + shiftKeys[Bytes[2]])
        else:

    print("[+] Found : %s" % (result))

    # clean the temp data
    os.system("rm ./%s" % (DataFileName))
